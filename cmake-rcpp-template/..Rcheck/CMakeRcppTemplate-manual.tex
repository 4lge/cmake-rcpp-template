\HeaderA{CLDistanceMatrix}{Distance Matrix Calculation via OpenCL}{CLDistanceMatrix}
\keyword{distance, opencl}{CLDistanceMatrix}
%
\begin{Description}
Given a tabular data set X calculates its distance matrix D, i.e.
D[i,j] = ||X[i,] - X[j,]||
via an OpenCL backend in C++.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
CLDistanceMatrix(mat)
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{mat}] A tabular dataset as a matrix
\end{ldescription}
\end{Arguments}
%
\begin{Value}
\begin{ldescription}
\item[\code{D}] Distance Matrix of X.
\end{ldescription}
\end{Value}
